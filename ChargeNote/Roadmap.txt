SELECT 
      Pays.[Title]
      ,[InstallmentDate]
      ,[Amount]
      ,Inst.[isPaid]
  FROM [MyOrganizer].[dbo].[InstallmentTable] as Inst
  LEFT JOIN PaymentTable as Pays on PaymentID = Pays.ID
  Where Month(InstallmentDate) = 10 and Year(InstallmentDate) = 2018
